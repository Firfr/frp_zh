  ```
    switch (状态) {
      case "running": return "运行中";
      case "error": return "错误";
      case "disabled": return "禁用";
      case "waiting": return "等待中";
      case "new": return "新创建";
      case "wait start": return "等待启动";
      case "start error": return "启动失败";
      case "check failed": return "检查失败";
      case "closed": return "已关闭";
      default: return 状态
    }
  }
  const fn代理错误信息转换 = (错误信息: string) =>{
    switch (错误信息) {
      case "port already used": return "端口已被占用" ;
      case "port not allowed": return "不允许使用该端口" ;
      case "port unavailable": return "端口不可用" ;
      case "no available port": return "没有可用端口" ;
      default: return 错误信息
    }
  }
  ```
