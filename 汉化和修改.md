#### 修改

修改默认提示`div.editor-wrapper>el-input` `placeholder` 的值。

```toml
'# frpc 配置文件参考内容...
# 文件格式为 TOML ，# 开头为注释

# 服务端IP或域名 根据实际修改
serverAddr = "127.0.0.1"
# 服务端通讯端口 根据实际修改
serverPort = 7000

# 用于服务端标识，不写会使用随机字符串，建议设置
#user = "用户名"
#clientID = "客户端ID"

# --------------- 认证 --------------- #

# 认证方式 以 token 认证方式为例
# 其他的认证方式参考官方文档
[auth]
# 认证方式
method = "token"
# token的值
token = "frps"

# ------------- 网页界面 ------------- #

# 非必须，建议设置

[webServer]
# 监听IP 不建议修改
addr = "0.0.0.0"
# 端口
port = 7400
# 用户名密码可以不设置，注释或删掉就行
user = "用户名"
password = "密码"
pprofEnable = false

# ------------- 日志配置 ------------- #

[log]
#to = "./frps.log" # 不写可以在容器的日志中查看
level = "info"     # 日志等级
#maxDays = 7     # 仅在日志写入文件时生效

# ------------ 转发配置说明 ------------ #

# 每一个 [[proxies]] 部分就是一个转发配置
# name 字段是唯一标识，是唯一的，不能重复
# transport.proxyProtocolVersion
#   用于获取真实访问地址取值 v1: 文本格式 v2: 二进制格式
#   http https 有时添加会无法访问
# enabled = false
#   作用是关闭这个配置快，不启用
#   这个的作用是 在不删除配置快的情况下关闭配置
#   注释或删除 enabled = false 启用配置
# 设置客户端与服务端之间数据压缩，默认关闭
#   transport.useEncryption = true
# 设置客户端与服务端之间数据加密，默认关闭
#   HTTPS 转发类型没必要开启，因为已经加密了
#   transport.useCompression = true

# ----------- 通用代理配置参考 ----------- #

[[proxies]]
name = "通用示例"
type = "tcp"
localIP = "本地 IP"
localPort = 本地端口
remotePort = 远程端口不写动态分配

# -------- HTTP HTTPS 代理配置参考 -------- #

# 代理类型 HTTP HTTPS
# 这种代理类型需要提前绑定域名到服务端
# 通过不同的域名区分不同的代理
# 远程端口只能通过在服务端添加配置指定
#   HTTP 端口 vhostHTTPPort
#   HTTPS 端口 vhostHTTPSPort
# 不能指定远程端口，只能设置子域名会域名
# 在开启子域名模式后，自定义的域名不能属于子域名
#   customDomains = ["域名"]

# ------------- HTTP-示例 ------------- #

[[proxies]]
enabled = false
name = "HTTP-示例"
type = "http"
localIP = "本地IP"
localPort = 本地端口
subdomain = "子域名"

# ------------- HTTPS-示例 ------------- #

[[proxies]]
enabled = false
name = "HTTPS-示例"
type = "https"
localIP = "本地IP"
localPort = 本地端口要有证书
customDomains = ["域名"]

'
```
