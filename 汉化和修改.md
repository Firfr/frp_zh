#### 修改

- 默认提示
  - 修改 `div.editor-wrapper>el-input` `placeholder` 的值
  ```toml
  '# frpc 配置文件参考内容...
  # 文件格式 TOML ，# 开头为注释
  
  [common]
  server_addr = 127.0.0.1   # 服务端地址
  server_port = 7000        # 服务端端口

  # 认证方式 以 token 方式为例
  # 其他认证方式配置 请阅读官方文档
  #auth.method = "认证方式"  # 与服务端的认证方式
  #auth.token = "令牌"      # token 认证方式时 的认证令牌
  
  # 用于服务端标识 下面两行不写时，服务端会生产随机字符串
  user = "用户名"
  clientID = "客户端ID"
  
  # 网页界面
  webServer.addr = "0.0.0.0"     # 界面地址
  webServer.port = 7140          # 界面端口
  webServer.pprofEnable = false  # 是否开启 pprof
  webServer.user = "用户名"      # 界面用户名
  webServer.password = "密码"    # 界面密码

  
  ## 代理示例
  #[[proxies]]            # 配置代理快
  #name = "测试"          # 代理名称 唯一
  #type = "tcp"           # 代理类型 无特殊情况填写 tcp
  #localIP = "127.0.0.1"  # 本地 IP
  #localPort = 2345       # 本地 端口
  #remotePort = 1212      # 远程端口，不写动态分配

  '
  ```
- 配置示例
  - 修改 `div.help-section>pre.code-example` 的值
  ```toml
  # 代理类型 HTTP
  # 这种代理类型需要提前绑定域名到服务端
  # 远程端口在服务端通过 vhostHTTPPort 指定
  # 通过不同的子域名区分不同的代理
  [[proxies]]
  name = "web"
  type = "http"
  localPort = 80     # 本地端口
  customDomains = ["example.com"]  # 要访问的域名

  # 通用代理配置参考
  [[proxies]]            # 配置代理快
  name = "示例"          # 代理名称 唯一
  type = "tcp"           # 代理类型 无特殊情况填写 tcp
  localIP = "127.0.0.1"  # 本地 IP
  localPort = 2345       # 本地 端口
  remotePort = 1212      # 远程端口，不写动态分配
  ```
